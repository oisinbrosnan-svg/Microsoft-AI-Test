    function drawTrackerChart() {
      const ctx = trackerChartCanvas.getContext("2d");
      const width = trackerChartCanvas.width;
      const height = trackerChartCanvas.height;
      ctx.clearRect(0, 0, width, height);

      const now = new Date();
      const days = [];
      for (let i = 6; i >= 0; i--) {
        const d = new Date(now);
        d.setDate(now.getDate() - i);
        days.push(d.toDateString());
      }

      const minutesPerDay = days.map(d =>
        data.sessions
          .filter(s => new Date(s.createdAt).toDateString() === d)
          .reduce((sum, s) => sum + s.duration, 0)
      );

      const maxMinutes = Math.max(30, ...minutesPerDay);
      const barWidth = width / (days.length * 1.5);
      const gap = barWidth / 2;

      minutesPerDay.forEach((m, idx) => {
        const x = idx * (barWidth + gap) + gap;
        const barHeight = (m / maxMinutes) * (height - 20);
        const y = height - barHeight - 10;
        const gradient = ctx.createLinearGradient(x, y, x, y + barHeight);
        gradient.addColorStop(0, "#4ea3ff");
        gradient.addColorStop(1, "#123c7c");
        ctx.fillStyle = gradient;
        ctx.fillRect(x, y, barWidth, barHeight);
        ctx.fillStyle = "#a9b3d1";
        ctx.font = "9px system-ui";
        ctx.textAlign = "center";
        ctx.fillText(m.toString(), x + barWidth / 2, y - 2);
      });

    function startPomodoro() {
      if (pomodoroRunning) return;
      pomodoroRunning = true;
      if (pomodoroSecondsRemaining <= 0) {
        pomodoroOnBreak = false;
        pomodoroSecondsTotal = data.settings.pomodoroMinutes * 60;
        pomodoroSecondsRemaining = pomodoroSecondsTotal;
      }
      if (pomodoroInterval) clearInterval(pomodoroInterval);
      pomodoroInterval = setInterval(() => {
        pomodoroSecondsRemaining--;
        if (pomodoroSecondsRemaining <= 0) {
          pomodoroSecondsRemaining = 0;
          pomodoroRunning = false;
          clearInterval(pomodoroInterval);
          if (!pomodoroOnBreak) {
            alert("Nice work. Take a gentle 5-minute break.");
            pomodoroOnBreak = true;
            pomodoroSecondsTotal = data.settings.breakMinutes * 60;
            pomodoroSecondsRemaining = pomodoroSecondsTotal;
          } else {

            pomodoroOnBreak = false;
            pomodoroSecondsTotal = data.settings.pomodoroMinutes * 60;
            pomodoroSecondsRemaining = pomodoroSecondsTotal;
          }
        }
        updatePomodoroDisplay();
      }, 1000);
      updatePomodoroDisplay();
    }

        : "No sessions logged yet. Whenever you‚Äôre ready, even a 5-minute session counts.";
    }
